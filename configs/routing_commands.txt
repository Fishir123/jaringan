# routing_commands.txt — Modul 4 (srv01 sebagai router)
# Kelompok X=2. LAN-A 10.10.2.0/24, LAN-B 10.20.2.0/24.

## srv01 — interface LAN-B (enp0s9) statik (ifupdown)
# /etc/network/interfaces.d/lanB
auto enp0s9
iface enp0s9 inet static
    address 10.20.2.1
    netmask 255.255.255.0
# apply:
sudo ifup enp0s9   # atau: sudo systemctl restart networking

## srv01 — enable IP forwarding (permanen)
echo "net.ipv4.ip_forward=1" | sudo tee /etc/sysctl.d/99-router.conf
sudo sysctl --system
cat /proc/sys/net/ipv4/ip_forward   # harus 1

## cli01 — pindah ke LAN-B + gateway
# /etc/network/interfaces.d/lan0
auto enp0s8
iface enp0s8 inet static
    address 10.20.2.2
    netmask 255.255.255.0
    gateway 10.20.2.1
sudo systemctl restart networking

## adm01 — static route ke LAN-B via srv01 (LAN-A)
sudo ip route add 10.20.2.0/24 via 10.10.2.11
# persisten via /etc/network/interfaces.d/lan0:
#   post-up ip route add 10.20.2.0/24 via 10.10.2.11 || true

## Uji konektivitas antar-subnet
# dari adm01:
ping 10.20.2.2          # adm01 (LAN-A) -> cli01 (LAN-B), TTL=63 (1 hop)
# dari cli01:
ping 10.10.2.10         # cli01 -> adm01 via gateway srv01

## Packet analysis (tcpdump di srv01)
sudo tcpdump -i enp0s8 -n icmp                 # lihat echo request/reply
sudo tcpdump -i any -n tcp port 22 and host 10.20.2.2   # lihat SYN/SYN-ACK/ACK

## NAT (bonus) — agar LAN-B akses internet via srv01
sudo apt install nftables
sudo nft add table ip nat
sudo nft add chain ip nat postrouting { type nat hook postrouting priority 100 \; }
sudo nft add rule ip nat postrouting oifname "enp0s3" masquerade
